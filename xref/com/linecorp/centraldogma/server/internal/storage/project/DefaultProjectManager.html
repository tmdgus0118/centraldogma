17  package com.linecorp.centraldogma.server.internal.storage.project;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import java.io.File;
22  import java.util.concurrent.Executor;
23  import java.util.function.Supplier;
24  
25  import javax.annotation.Nullable;
26  
27  import com.github.benmanes.caffeine.cache.stats.CacheStats;
28  
29  import com.linecorp.centraldogma.common.Author;
30  import com.linecorp.centraldogma.common.CentralDogmaException;
31  import com.linecorp.centraldogma.common.ProjectExistsException;
32  import com.linecorp.centraldogma.common.ProjectNotFoundException;
33  import com.linecorp.centraldogma.server.internal.storage.DirectoryBasedStorageManager;
34  import com.linecorp.centraldogma.server.internal.storage.repository.cache.RepositoryCache;
35  
36  public class DefaultProjectManager extends DirectoryBasedStorageManager<Project> implements ProjectManager {
37  
38      @Nullable
39      private final RepositoryCache cache;
40  
41      public DefaultProjectManager(File rootDir, Executor repositoryWorker, @Nullable String cacheSpec) {
42          super(rootDir, Project.class,
43                requireNonNull(repositoryWorker, "repositoryWorker"),
44                cacheSpec != null ? new RepositoryCache(cacheSpec) : null);
45          cache = (RepositoryCache) childArg(1);
46      }
47  
48      @Override
49      public CacheStats cacheStats() {
50          return cache != null ? cache.stats() : CacheStats.empty();
51      }
52  
53      @Override
54      public void close(Supplier<CentralDogmaException> failureCauseSupplier) {
55          super.close(failureCauseSupplier);
56          if (cache != null) {
57              cache.clear();
58          }
59      }
60  
61      @Override
62      protected Project openChild(File childDir, Object[] childArgs) throws Exception {
63          return new DefaultProject(childDir, (Executor) childArgs[0], (RepositoryCache) childArgs[1]);
64      }
65  
66      @Override
67      protected Project createChild(File childDir, Object[] childArgs,
68                                    Author author, long creationTimeMillis) throws Exception {
69          return new DefaultProject(childDir, (Executor) childArgs[0], (RepositoryCache) childArgs[1],
70                                    creationTimeMillis, author
71          );
72      }
73  
74      @Override
75      protected void closeChild(File childDir, Project child,
76                                Supplier<CentralDogmaException> failureCauseSupplier) {
77          final DefaultProject c = (DefaultProject) child;
78          c.repos.close(failureCauseSupplier);
79      }
80  
81      @Override
82      protected CentralDogmaException newStorageExistsException(String name) {
83          return new ProjectExistsException(name);
84      }
85  
86      @Override
87      protected CentralDogmaException newStorageNotFoundException(String name) {
88          return new ProjectNotFoundException(name);
89      }
90  }
